   270→  // Seed categories
   271→  for (const cat of CATEGORIES) {
   272→    await db.category.upsert({
   273→      where: { slug: cat.slug },
   274→      update: { name: cat.name, icon: cat.icon, description: cat.description, order: cat.order },
   275→      create: { id: cat.id, name: cat.name, slug: cat.slug, icon: cat.icon, description: cat.description, order: cat.order },
   276→    });
   277→  }
   278→
   279→  // Seed users
   280→  for (const user of SEED_USERS) {
   281→    await db.user.upsert({
   282→      where: { email: user.email },
   283→      update: { name: user.name, role: user.role, company: user.company, country: user.country, city: user.city, verified: user.verified },
   284→      create: user,
   285→    });
   286→  }
   287→
   288→  // Seed projects
   289→  for (const proj of SEED_PROJECTS) {
   290→    const existing = await db.project.findUnique({ where: { id: proj.id } });
   291→    if (!existing) {
   292→      await db.project.create({ data: proj as any });
   293→    }
   294→  }
   295→
   296→  return { categories: CATEGORIES.length, users: SEED_USERS.length, projects: SEED_PROJECTS.length };
